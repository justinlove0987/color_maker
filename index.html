    <script>

        // 定義顏色列表

        const colors = ['red', 'blue', 'yellow', 'white'];


        function shuffleArray(array) {

            // 洗牌演算法，隨機排列陣列

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

        }


        function generateRandomColors() {

            // 清空目前顯示的顏色方塊

            document.body.innerHTML = '';


            // 複製顏色陣列，並洗牌

            const shuffledColors = [...colors];

            shuffleArray(shuffledColors);


            // 增加顏色數量

            const displayedColors = shuffledColors.concat(shuffledColors.slice(0, 3));


            // 隨機排序顏色後顯示

            shuffleArray(displayedColors);

            displayedColors.forEach(color => {

                const colorBox = document.createElement('div');

                colorBox.className = `color-box ${color}`;

                colorBox.textContent = color.charAt(0).toUpperCase() + color.slice(1);

                document.body.appendChild(colorBox);

            });

        }


        // 初始顯示

        generateRandomColors();


        // 每5秒刷新一次顏色

        setInterval(generateRandomColors, 5000);

    </script>
